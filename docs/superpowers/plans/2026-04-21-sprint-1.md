# Sprint 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the full Sprint 1 loop — facilitator creates a session, participants join via QR, drag a dot up/down, position writes to Google Sheets every 2s, facilitator stops the session.

**Architecture:** Google Apps Script Web App (GET-only, all actions via `?action=` param) is the sole backend. Three plain HTML pages (facilitator, join, participant) talk to it directly. Master Google Sheet stores a session registry; each session gets its own fresh Google Sheet.

**Tech Stack:** Plain HTML/CSS/JS (no framework, no build), Google Apps Script, Google Sheets API (via Apps Script), qrcode.js CDN, Vercel (static hosting + URL rewrite)

```javascript
var REGISTRY_ID = '1t35mJ8TIY60ePOrEgTPZ_oGUX69atXMhAdQmYS8Ralo';

var WORDS = [
  'DRUM', 'MILO', 'APEX', 'COVE', 'FLINT', 'GROVE', 'HALO', 'INK', 'JADE', 'KITE',
  'LARK', 'MESH', 'NOON', 'OAK', 'PINE', 'QUAY', 'REEF', 'SAGE', 'TIDE', 'URN',
  'VALE', 'WAVE', 'YARN', 'ZINC', 'BOLT', 'CALM', 'DUSK', 'ECHO', 'FERN', 'GUST',
  'HIVE', 'IRIS', 'JEST', 'KELP', 'LIME', 'MAST', 'NOOK', 'OPAL', 'PEAT', 'QUILL',
  'RUNE', 'SALT', 'TARN', 'UMBER', 'VANE', 'WREN', 'YOKE', 'ZEST', 'BIRCH', 'CLAY'
];

function doGet(e) {
  var p = e.parameter;
  var result;
  try {
    switch (p.action) {
      case 'create':  result = createSession(); break;
      case 'resolve': result = resolveCode(p.code); break;
      case 'start':   result = setStatus(p.sheetId, 'ACTIVE'); break;
      case 'stop':    result = setStatus(p.sheetId, 'ENDED'); break;
      case 'poll':    result = pollStatus(p.sheetId); break;
      case 'log':     result = logPoint(p.sheetId, p.participant, parseFloat(p.value)); break;
      default:        result = { error: 'unknown action' };
    }
  } catch (err) {
    result = { error: err.message };
  }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function createSession() {
  var registry = SpreadsheetApp.openById(REGISTRY_ID).getSheetByName('registry');
  var values = registry.getDataRange().getValues();
  var usedCodes = values.slice(1).map(function(r) { return r[0]; });
  var available = WORDS.filter(function(w) { return usedCodes.indexOf(w) === -1; });
  if (available.length === 0) return { error: 'no codes available' };

  var code = available[Math.floor(Math.random() * available.length)];
  var now = new Date().toISOString();

  var ss = SpreadsheetApp.create('The Room — ' + code);
  var sheet = ss.getSheets()[0];
  sheet.setName(code);
  sheet.appendRow(['type', 'participant', 'timestamp', 'value']);
  sheet.appendRow(['STATUS', '', now, 'WAITING']);

  registry.appendRow([code, ss.getId(), 'WAITING', now]);
  return { code: code, sheetId: ss.getId() };
}

function resolveCode(code) {
  var registry = SpreadsheetApp.openById(REGISTRY_ID).getSheetByName('registry');
  var values = registry.getDataRange().getValues();
  for (var i = 1; i < values.length; i++) {
    if (values[i][0] === code) {
      return { sheetId: values[i][1], status: values[i][2] };
    }
  }
  return { error: 'not found' };
}

function pollStatus(sheetId) {
  var sheet = SpreadsheetApp.openById(sheetId).getSheets()[0];
  var data = sheet.getDataRange().getValues();
  // data[0] = header row, data[1] = STATUS row ['STATUS', '', timestamp, 'WAITING']
  var status = data[1][3];
  var participants = {};
  for (var i = 2; i < data.length; i++) {
    if (data[i][0] === 'DATA' && data[i][1]) {
      participants[data[i][1]] = true;
    }
  }
  return { status: status, count: Object.keys(participants).length };
}

function setStatus(sheetId, status) {
  var sheet = SpreadsheetApp.openById(sheetId).getSheets()[0];
  // Row 2 (STATUS row, 1-indexed), col 4 (value column, 1-indexed)
  sheet.getRange(2, 4).setValue(status);

  var registry = SpreadsheetApp.openById(REGISTRY_ID).getSheetByName('registry');
  var values = registry.getDataRange().getValues();
  for (var i = 1; i < values.length; i++) {
    if (values[i][1] === sheetId) {
      registry.getRange(i + 1, 3).setValue(status);
      break;
    }
  }
  return { ok: true };
}

function logPoint(sheetId, participant, value) {
  var sheet = SpreadsheetApp.openById(sheetId).getSheets()[0];
  sheet.appendRow(['DATA', participant, new Date().toISOString(), value]);
  return { ok: true };
}
```

        s.classList.remove('active');
      });
      document.getElementById('screen-' + name).classList.add('active');

      var el = document.getElementById('qr');
      while (el.firstChild) el.removeChild(el.firstChild);

      var btn = document.getElementById('btn-create');
      btn.textContent = 'Creating…';
      btn.disabled = true;
      api({ action: 'create' }).then(function(data) {
        if (data.error) {
          alert('Error: ' + data.error);
          btn.textContent = 'Create session';
          btn.disabled = false;
          return;

        localStorage.setItem('theroom_session', JSON.stringify(session));
        showWaiting(0);

      document.getElementById('code-waiting').textContent = session.code;
      var joinUrl = window.location.origin + '/join/' + session.code;
      document.getElementById('url-waiting').textContent = joinUrl;
      clearQr();
      new QRCode(document.getElementById('qr'), {
        text: joinUrl,
        width: 160,
        height: 160,
        colorDark: '#000000',
        colorLight: '#ffffff'
      });
      document.getElementById('count-waiting').textContent = count;
      showScreen('waiting');
      startPolling();
    }

    function showRunning(count) {
      document.getElementById('code-running').textContent = session.code;
      document.getElementById('count-running').textContent = count;
      showScreen('running');
      startPolling();
    }

    function startPolling() {
      clearInterval(pollTimer);
      pollTimer = setInterval(function() {
        api({ action: 'poll', sheetId: session.sheetId }).then(function(data) {
          if (data.error) return;
          if (data.status === 'WAITING') {
            document.getElementById('count-waiting').textContent = data.count || 0;
          } else if (data.status === 'ACTIVE') {
            document.getElementById('count-running').textContent = data.count || 0;
          }
        });
      }, 2000);
    }

    function startSession() {
      api({ action: 'start', sheetId: session.sheetId }).then(function(data) {
        if (data.error) { alert('Error: ' + data.error); return; }
        var count = parseInt(document.getElementById('count-waiting').textContent) || 0;
        showRunning(count);
      });
    }

    function stopSession() {
      if (!confirm('Stop the session? This cannot be undone.')) return;
      clearInterval(pollTimer);
      api({ action: 'stop', sheetId: session.sheetId }).then(function() {
        localStorage.removeItem('theroom_session');
        session = null;
        var btn = document.getElementById('btn-create');
        btn.textContent = 'Create session';
        btn.disabled = false;
        showScreen('create');
      });
    }

    // Restore from localStorage on page load
    var saved = localStorage.getItem('theroom_session');
    if (saved) {
      session = JSON.parse(saved);
      api({ action: 'poll', sheetId: session.sheetId }).then(function(data) {
        if (data.error || data.status === 'ENDED') {
          localStorage.removeItem('theroom_session');
          return;
        }
        if (data.status === 'ACTIVE') showRunning(data.count || 0);
        else showWaiting(data.count || 0);
      });
    }
  </script>
</body>
</html>

    var pollTimer = null;

    document.getElementById('session-code').textContent = session.code;
    document.getElementById('participant-name').textContent = session.name;

    function api(params) {
      return fetch(SCRIPT_URL + '?' + new URLSearchParams(params).toString())
        .then(function(r) { return r.json(); });

    function updateDot(v) {
      value = Math.max(0, Math.min(1, v));
      document.getElementById('dot').style.top = ((1 - value) * 100) + '%';
      document.getElementById('fill').style.height = (value * 100) + '%';
    }

    function startActive() {
      if (active) return;
      active = true;
      document.getElementById('session-indicator').style.background = 'rgba(255,255,255,0.45)';
      logTimer = setInterval(function() {
        api({
          action: 'log',
          sheetId: session.sheetId,
          participant: session.name,
          value: value.toFixed(3)
        });
      }, 2000);
    }

    function endSession() {
      active = false;
      clearInterval(logTimer);
      clearInterval(pollTimer);
      document.getElementById('field').style.display = 'none';

        if (data.status === 'ENDED') endSession();
      });
